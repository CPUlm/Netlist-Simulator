208/// Possible flags for a register.
209/// \see RegisterInfo
210enum RegisterInfoFlag {
211 RIF_NONE = 0x0,
212 /// The register represents an input.
213 RIF_INPUT = 0x1,
214 /// The register represents an output.
215 RIF_OUTPUT = 0x2,
216 /// The register is an internal register used by the parser to implement
217 /// some functionalities. It doesn't correspond to a variable declared
218 /// in the `VAR` statement.

220};

222/// Meta information about a program's register.

228 bus_size_t bus_size = 1;
229 /// \see RegisterInfoFlag
