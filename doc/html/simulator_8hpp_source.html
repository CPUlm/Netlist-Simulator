32 /// \brief Prepares the given Netlist program for further simulation.
33 ///
34 /// This function may be used to compile the given program to machine code
35 /// or do any optimizations for an interpreter.
36 ///
37 /// \param program
38 /// \return False in case of failure.
39 virtual bool prepare(const std::shared_ptr<Program> &program) = 0;
40 /// \brief Simulates a Netlist program.
41 ///
42 /// How the Netlist program is effectively simulated is implementation defined.
43 /// Internally, the program may be interpreted or compiled to machine code and then
44 /// executed. The only thing important is that for the same program and the
45 /// same inputs you should always get the same outputs.
46 ///
47 /// \param inputs The inputs of the Netlist, in the same order as they were found in the program.
48 /// \param outputs The outputs of the Netlist, in the same order as they were found in the program.
49 /// \param n The count of cycles to simulate. By default is 1 cycle.
50 virtual void simulate(reg_value_t *inputs, reg_value_t *outputs, size_t n = 1) = 0;
51
52 // ------------------------------------------------------
53 // The debugger API
54 // ------------------------------------------------------
55
56 /// \brief Does this backend support the debugger API of the simulator?
57 ///
58 /// By default, this function returns false.
59 [[nodiscard]] virtual bool has_debugger() const { return false; }
60
61 /// \brief Returns the value of the given requested \a reg.
62 ///
63 /// \warning This function only works if has_debugger() returns true.
64 ///
65 /// \param reg The register to request. If \a reg does not exists, the behavior is undefined.
66 /// \return The bits of the requested register stored in the lowest bit of the returned value.
67 [[nodiscard]] virtual reg_value_t get_register(reg_t reg) const { assert(false && "debugger API not supported"); return 0; }
68 /// \brief Sets the value of the given requested \a reg to \a value.
69 ///
70 /// \warning This function only works if has_debugger() returns true.
71 ///
72 /// \param reg The register to update. If \a reg does not exists, the behavior is undefined.
73 /// \param value The new register's value. Only the lowest bits of \a value that can be
74 /// stored in \a reg are considered. Others are discarded.
75 virtual void set_register(reg_t reg, reg_value_t value) { assert(false && "debugger API not supported"); }
76
77 /// \brief Simulates a single instruction in the Netlist program.
78 ///
79 /// \warning This function only works if has_debugger() returns true.
80 virtual void step() { assert(false && "debugger API not supported"); }
