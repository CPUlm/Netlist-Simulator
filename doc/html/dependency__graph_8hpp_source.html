11/// \brief Represents a dependency graph for the a Netlist program.

13/// Notably, it is in this class that the scheduling is done.
14///
15/// For example:
16/// ```
17/// std::shared_ptr<Program> program = ...;
18/// DependencyGraph graph = DependencyGraph::build(program);
19/// graph.schedule();
20/// // program is now correctly scheduled
21/// ```
22class DependencyGraph {
23public:
24 /// Builds the dependency graph for the given program.

27 /// Returns true if \a from depends on \a to.
28 [[nodiscard]] bool depends(reg_t from, reg_t to) const;
29
30 /// Reorder the instructions of the graph's program so all all dependencies
31 /// are respected. This function corresponds to the first questions of the Tutorial.
32 void schedule(ReportManager& report_manager);
33
34 /// Same as dump_dot(std::ostream&) with the std::cout argument.
35 void dump_dot();

39
40private:
41 explicit DependencyGraph(const std::shared_ptr<Program> &program);
42
43 /// Adds a dependency between two registers.
